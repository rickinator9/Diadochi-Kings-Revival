# Gain one of a vassal's landed titles
plot_win_silver_shields = {
	# Plotter scope
	potential = {
		is_playable = yes
		prisoner = no
		is_patrician = no
		is_merchant_republic = no
		is_adult = yes
		culture_group = hellenic
		primary_title = {
			higher_tier_than = DUKE
			temporary = no
		}
		NOT = { trait = incapable }
		OR = {
			ai = no
			war = no
		}
		d_silver_shields = {
			has_holder = yes
		}
		NOT = {
			any_vassal = {
				has_landed_title = d_silver_shields
			}
		}
	}
	
	target_potential = {
		has_landed_title = d_silver_shields
	}
	
	allow = {
		NOT = {
			FROM = {
				any_vassal = {
					has_landed_title = d_silver_shields
				}
			}
		}
	}
	
	chance = {
		factor = 1
		modifier = {
			factor = 0.0
			FROM = {
				ai = yes
				war = yes
			}
		}
		modifier = {
			factor = 0.01
			FROM = { trait = content }
		}
		modifier = {
			factor = 0.2
			FROM = {
				OR = {
					trait = dull
					trait = slow
				}
			}
		}
		modifier = {
			factor = 0.2
			FROM = { trait = kind }
		}
		modifier = {
			factor = 4.0
			FROM = { trait = ambitious }
		}
		modifier = {
			factor = 4.0
			FROM = { trait = ambitious }
		}
	}
	
	success = {
		FROM = {
			any_vassal = {
				has_landed_title = d_silver_shields
			}
		}
	}
	abort = {
		FROM = {
			OR = {
				is_playable = no
				trait = incapable
				in_revolt = yes
				has_landed_title = d_silver_shields
			}
		}
	}
	abort_effect = {
	}
	effect = {
		FROM = {
			any_plot_backer = {
				reverse_opinion = {
					modifier = opinion_grateful
					who = FROM
					years = 5
				}
			}
			# Clear flags
			hidden_tooltip = {
				clr_character_flag = event7040
				clr_character_flag = plot_gain_vassal_title_decision_taken
			}
		}
	}
}
