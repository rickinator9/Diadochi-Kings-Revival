regency_war_cb = {
	name = CB_NAME_REGENCY_WAR
	war_name = WAR_NAME_REGENCY_WAR
	sprite = 9
	truce_days = 7300
	press_claim = yes
	is_permanent = yes
	hostile_against_others = yes
	can_ask_to_join_war = no
	battle_warscore_mult = 2.0

	allowed_to_target_tributaries = no
	defender_can_call_allies = no
	can_call_vassals = yes

	max_attacker_battle_score = 100
	max_defender_battle_score = 100
	infamy_modifier = 0.15

	allow_distant = yes

	can_use_gui = {
		ROOT = {
			is_ruler = yes
			higher_real_tier_than = DUKE
			NOT = {
				independent = yes
			}
			NOT = {
				has_minor_title = title_regent
			}
			government = satrap_government
			same_realm = FROM
		}
	}

	can_use = {
		FROM = {
			is_ruler = yes
			higher_real_tier_than = DUKE
			NOT = {
				independent = yes
			}
			has_minor_title = title_regent
			same_realm = ROOT
		}
	}

	is_valid = {
		ROOT = {
			is_ruler = yes
			higher_real_tier_than = DUKE
			NOT = {
				independent = yes
			}
			NOT = {
				has_minor_title = title_regent
			}
			same_realm = FROM
		}
		FROM = {
			is_ruler = yes
			higher_real_tier_than = DUKE
			NOT = {
				independent = yes
			}
			has_minor_title = title_regent
			same_realm = ROOT
		}
	}

	on_success = {
		FROM = {
			prestige = -100
			remove_title = title_regent
		}
		ROOT = {
			prestige = 100
			give_minor_title = title_regent
		}
	}

	on_fail = {
		ROOT = {
			prestige = -100
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -100
			transfer_scaled_wealth = {
				to = FROM
				value = 2
				min = 100
			}
			top_liege = {
				opinion = {
					modifier = grievously_offended
					who = ROOT
				}
			}
		}
		FROM = {
			prestige = 100
			top_liege = {
				opinion = {
					modifier = grievously_offended
					who = ROOT
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}
